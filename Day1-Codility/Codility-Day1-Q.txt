void FindFactorials(int num1, int num2)
        {
            int count = 0;
            for (int i = num1; i <= num2; i++)
            {
                int n = i;
                for (int j = 2; n == 1 || j <= i; j++)
                {
                    if (n % j == 0)
                        n /= j;
                    else
                        break;
                }
                if (n == 1)
                    count++;
            }
            Console.WriteLine(count);
        }

public int solution(int N) {
       int len =0,maxLen=0;
       bool started = false;
       for(uint i=1;i<=N;i<<=1)
       {
           if((i&N) !=0)
           {
               if(started)
               {
                   if(len >maxLen)
                        maxLen = len;
               len = 0;
               }
               started = true;
           }
           else if(started)
                len++;
       }
       return maxLen;
       
    }
